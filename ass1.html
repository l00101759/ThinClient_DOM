function goToRightSibling()
{
	$check = $curr.parent();
	$check = $check.next(); 

	if($check.length)
	{
		$curr.css( "background", "white" );
		$curr = $curr.parent();
		$curr = $curr.next(); 
		$curr = $curr.find("img:first");
		$curr.css( "background", "#f99" );
	}

}





function goToLeftSibling()
{
	$check = $curr.parent();
	$check = $check.prev(); 

	if($check.length)
	{
		$curr.css( "background", "white" );
		$curr = $curr.parent(); 
		$curr = $curr.prev(); 
		$curr = $curr.find("img:first");
		$curr.css( "background", "#f99" );
	}
}





function goToParent()
{

	}

}
